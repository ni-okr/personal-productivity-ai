const cache = new Map()

export function memoize<T>(fn: Function, ttl: number = 300000): T {
  return ((...args: any[]) => {
    const key = JSON.stringify(args)
    const cached = cache.get(key)
    
    if (cached && Date.now() - cached.timestamp < ttl) {
      return cached.value
    }

    return result
  }) as T
}
